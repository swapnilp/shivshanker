def teams
  team_ids = []

  games.each do |g|
    team_ids << g.home_team_id
    team_ids << g.away_team_id
  end

  team_ids.uniq!

  Team.joins(:game_teams).where(:game_teams => {:team_id => team_ids }).group(:team_id)
end
