def eligible? user
  cu = contest_users.where(:user_id => user.id).first
  
  if cu.nil?
    ContestUser.transaction do
      cu = ContestUser.new
      cu.contest = self
      cu.user = user
      cu.save!
    end
  end

  return cu.eligible
end
